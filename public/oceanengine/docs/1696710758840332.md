# 基础实践：频控限流器开发

本方案结合滑动窗口限流器与消息队列消费机制，实现对API接口的QPS精准控制，统计规则采用动态时间窗口（当前请求发起时刻倒推1秒），主要技术特点：

📌 核心设计理念：通过滑动窗口算法保证统计绝对时间精度，结合消息队列实现流量削峰填谷

```
// 时间窗口分片结构
type TimeSlice struct {
timestamp int64 // 精确到毫秒的时间标识
count     int64 // 该毫秒内的请求计数
}
​
// 滑动窗口限流器
type SlidingWindowLimiter struct {
windowDuration int64   // 窗口总时长(单位：毫秒)
maxRequests    int64   // 窗口内允许的最大请求数
slices         []*TimeSlice // 动态时间片集合（非严格环形）
currentPos     int     // 最新时间片写入位置指针
mutex          sync.Mutex // 并发控制锁
}
​
```

统计逻辑：

```
xxxxxxxxxx
func (l *SlidingWindowLimiter) Allow() bool {
l.mutex.Lock()
defer l.mutex.Unlock()
​
now := time.Now().UnixNano() / 1e6
// 清理过期时间片
cutoff := now - l.windowDuration
for i := 0; i < len(l.slices); {
if l.slices[i].timestamp < cutoff {
l.slices = append(l.slices[:i], l.slices[i+1:]...)
} else {
i++
}
}
​
// 统计当前窗口请求总数
var total int64
for _, s := range l.slices {
total += s.count
}
if total >= l.maxRequests {
return false
}
​
// 记录当前请求
if len(l.slices) > 0 && l.slices[l.currentPos].timestamp == now {
l.slices[l.currentPos].count++
} else {
l.slices = append(l.slices, &TimeSlice{
timestamp: now,
count:     1,
})
l.currentPos = len(l.slices) - 1
}
return true
}
​
```

## 消息队列消费实现

```
func StartConsumers(queue chan Message, limiter *SlidingWindowLimiter, workers int) {
for i := 0; i < workers; i++ {
go func(workerID int) {
for msg := range queue {
// 限流检查
if !limiter.Allow() {
// 进入弹性缓冲队列
select {
case bufferChan <- msg:
default:
// 缓冲队列满时转持久化存储
SendToRetryQueue(msg)
}
continue
}
// 执行API调用
resp, err := CallTargetAPI(msg)
if err != nil {
HandleAPIError(msg, err)
}
ProcessResponse(resp)
}
}(i)
}
}
```
