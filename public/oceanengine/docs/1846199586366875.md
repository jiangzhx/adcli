e.printStackTrace();

});

.url(url)
.method("GET", null)
.addHeader("Access-Token", ACCESS_TOKEN)
.build();

}

| data | object | json返回值 |
| list | list | json返回值 |
| id | int | 词包id |
| dpa_id | int | 商品库id |
| default_word | string | 默认词 |
